\begin{eg}
    
    Give an example of $ T, S \in L(V) $, $ V $ finite-dimensional such that $ TS = 0 $ but $ T,S \neq 0 $.

    \vspace{1em}
    \underbar{\textbf{Solution:}}
    $ \begin{cases}
        (T - I)(x,y) = (-x, y) - (x, y) = (-2x, 0) \\
        (T + I)(x, y) = (-x, y) + (x, y) = (0, 2y)
    \end{cases} $

    These are non-linear maps, but $ (T - I)(T + I) = 0 $

\end{eg}

\section{The Minimal Polynomial}
\begin{theorem}

    If $ V $ is a finite-dimensional vector space over $ \mathbb{C} $, and $ T \in L(V) $, then $ T $ has at least 1 eigenvalue.

    \vspace{1em}
    \begin{note}
        
        \[
            T(a_0, a_1, \cdots) = (0, a_0, a_1, \cdots )
        \]
        $ T \in L(\mathbb{C}^{\infty}) $ has no eigenvalue.

    \end{note}

\end{theorem}

\begin{proof} \leavevmode
    
    Let $ 0 \neq v \in V $, and $ n = \dim V $.
    
    Then look at the nth vector $ v, Tv, T^2v, \cdots , T^nv $.

    \vspace{1em}
    Since $ \dim V = n $, they're linearly dependent:
    \[
        a_0v + a_1Tv + \cdots + a_{n} T^n v = 0
    \]
    with not all the $ a_{i} = 0 $.

    \vspace{1em}
    Let $ p(x) = a_0 + a_1x + \cdots + a_{n}x^n \in P(\mathbb{C}) $.

    Then $ p(x) $ factors as
    \[
        p(x) = c(x - \lambda_1) \cdots (x - \lambda_{n})
    \]

    This is applying linear operator $ (T - \lambda_{n}I)\cdots (T - \lambda_1I) = p(T) $

    \vspace{1em}
    But \begin{align*}
        p(T)v 
        &= (a_0I + a_1T + \cdots + a_{n}T^n ) v \\
        &= a_0v + a_1Tv + \cdots + a_{n}T^n v \\
        &= 0
    \end{align*}

    $ v \neq 0 \implies p(T)$ is not injective. So there is $ \lambda_{i} $ such that $ T - \lambda_{i}I $ is not injective. So $ \lambda_{i} $ is an eigenvalue for $ T $.

\end{proof}

\section{Upper-Triangular Matrices}

\begin{definition}
    
    $ A_{n \times n} $ is an upper-triangular if all the entries below diagonal are 0.

\end{definition}

\begin{eg}
    
    $ 
    \begin{bmatrix}
        1 & 0 & 3 \\
        0 & 5 & -2 \\
        0 & 0 & -1 \\
    \end{bmatrix}
    $

\end{eg}
