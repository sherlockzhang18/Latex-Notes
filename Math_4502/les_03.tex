\subsection{}
\begin{itemize}
    \item Training data  $ D = \{(x_{n}, y_{n})\}_{n=1}^{\mathbb{N}} $
    \item 
        Parametric Model $ f(.; w) $
    
        $ \hat{y} = f(x; w) $ estimates $ y $ given $ x $.

        \begin{eg}
            $ f(x;w) = \sum_{k=1}^{n} w_{k}f_{k}(x)  $
        \end{eg}
    \item Minimized loss $ L(w) $
        \begin{eg}
            $ L(w) = \frac{1}{N}\sum_{n=1}^{N}\left\lVert y_{n} - f(x_{n}; w) \right\rVert ^2 $
        \end{eg}
\end{itemize}
