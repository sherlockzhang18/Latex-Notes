Want to know how many iterations to satisfy $ |x_k- \xi|  < \epsilon $ ?

\begin{itemize}
    \item Need $ L^k |x_0-\xi| < \epsilon $, \hspace{2em} $ k > \frac{1}{\log L} \cdot \log \frac{\epsilon}{|x_0-\xi|} $, for $ \epsilon < |x_0-\xi| $
    \item Not quite useful since $ \xi $ is unknown.
\end{itemize}

\vspace{2em}
\begin{definition} \underbar{\textbf{Cauchy Sequence}} \leavevmode
    
    \noindent $ (x_k)_{k = 0}^\infty $ is Cauchy if and only if for given $ \epsilon > 0 $, there is $ N \in \mathbb{N} $ such that $ n,m \geqslant \mathbb{N} \implies |x_n-x_m| < \epsilon$.
\end{definition}

\vspace{1em}
Cauchy Completeness: $ \mathbb{R} $ is "complete", i.e Cauchy Sequence $ (x_k)_{k = 0}^\infty  \subseteq \mathbb{R}$ converges to a limit in $ \mathbb{R} $.

\vspace{1em}
\setcounter{theorem}{2}
\begin{theorem} \leavevmode
    $ g: [a,b] \to [a,b] $ is a contraction. Simple iteration ($ x_{k+1} = g(x_k) $) converges to the fixed point $ \xi $.
\end{theorem}

\begin{proof} Another way to prove this with Cauchy \leavevmode
    
    Given $ n,m \in \mathbb{N} $, (WLOG, assume $ n<m $). 

    \vspace{1em}
    By Triangle Inequality
    \[
        |x_m-x_n| \leqslant |x_m-x_{m-1}| + |x_{m-1} - x_{m-2}| + \cdots + |x_{n+1} - x_{n}| 
    \]


    Note $ |x_m - x_{m-1}| \leqslant |g(x_{m-1}) - g(x_{m-2})| \leqslant L|x_{m-1}-x_{m-2}| $. Since $ g $ is a contraction, $ L|x_{m-1}-x_{m-2}| \leqslant  \cdots  \leqslant L^{m-n+1}|x_{n-1}-x_{n}| $

    \begin{align*}
        |x_m - x_n| & \leqslant L^{m-n-1}|x_{n+1} + x_{n}| + L^{m-n-2}|x_{n+1} - x_{n}| + \cdots  + |x_{n+1} - x_{n}| \\
        & \leqslant \left(\sum_{j = 0}^{m-n-1} L^j\right)|x_{n+1} - x_n| \\
        & \leqslant \left(\sum_{j = 0}^{\infty} L^j\right)|x_{n+1} - x_n| \\
        & \leqslant \left(\frac{1}{1-L}\right)|x_{n+1} - x_n| \\
    \end{align*}

    So the problem becomes give $\epsilon >0 $, find $ N \in \mathbb{N} $ such that $ |x_m-x_n| \leqslant \left(\frac{1}{1-L}\right) |x_{n+1} - x_n| < \epsilon$ for $ m,n > N $

    \begin{align*}
        |x_n-x_m| &\leqslant \left(\frac{1}{1-L}\right) |x_{n+1} - x_n| \\
        & \leqslant \left(\frac{1}{1-L}\right) L^n |x_1-x_{0}| \\
        & < \epsilon \hspace{2em} \text{Want}
    \end{align*}

    Pick $ n $ large enough : $ N \geqslant \frac{1}{\log L} \log\left( \frac{\epsilon(1-L)}{|x_1-x_0|}\right)$
\end{proof}

\begin{itemize}
    \item Suppose $ | g(x) - g(y) | \leqslant L|x-y| $ for all $ x, y \in [a,b] $.
    
        If $ g $ was differentiable in $ (a,b) \hspace{1em} \lim_{y \to x} \frac{g(x)-  g(y)}{|x-y|} \leqslant L$, so $ |g'(x)| \leqslant L $ for all $ x \in (a,b) $.

    \item Conversely, if $ g $ was differentiable in $ (a,b) $ and $ |g'(x)| \leqslant L $ for all $ x \in(a,b) $, then 
    \[
        g(x) - g(y) = g'(n)(x-y)
    \] for some $ n $ between $ x $ and $ y $, and 
    \[
        |g(x) - g(y)| \leqslant |g'(n)| |x-y| \leqslant L|x-y| \hspace{1em} \text{ for all } x, y \in (a,b)
    \]

    \item If $ |g'(x)| \leqslant L < 1 $ then $ g $ is a contraction (for \underbar{all} x).
\end{itemize}
