\begin{theorem}
    $ A \in \mathbb{R}^{n \times n} $

    \begin{itemize}
        \item There is a diagonal entry $ a_{rr} $ such that $ |a_{rr} - a_{ii}| > \delta $ for $ i = 1, \cdots , n, i \neq r $
        \item $ |a_{ij} < \epsilon| $ for $ i, j = 1, \cdots , n , i \neq j$ and $ \epsilon < \frac{\delta}{2(n - 1)} $
    \end{itemize}
    then this implies that there is an eigenvalue $ \lambda $ of $ A $ such that
    \[
        |a_{rr} - \lambda| \leqslant \frac{2(n-1)}{\delta}\epsilon^2
    \]
\end{theorem}
