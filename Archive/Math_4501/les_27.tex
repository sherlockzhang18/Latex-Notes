\begin{theorem}
    
    $ A \in \mathbb{R}^{n \times n}_{\text{sym}} $, $ \left\lVert \frac{x}{\left\lVert x \right\rVert _2} - v_{i} \right\rVert _2 \leqslant C \epsilon $ for some constant $ C $, then
    \[
        \left|R\left(\frac{x}{\left\lVert x \right\rVert _2} - \lambda_i\right)\right| \leqslant D\epsilon^2
    \]

\end{theorem}

\begin{proof}
    
    $ \{v_i\}_{i=1}^{n} $ forms orthonormal basis of eigenvectors.

    \vspace{1em}
    $ c_{i} = v_1^T \frac{x}{\left\lVert x \right\rVert _2} $

    \begin{align*}
        \left\lVert \frac{x}{\left\lVert x \right\rVert _2} - v_{i} \right\rVert _2^2 
        &= \left(\frac{x}{\left\lVert x \right\rVert _2} - v_{i}\right) ^T \left(\frac{x}{\left\lVert x \right\rVert _2} - v_{i}\right) \\
        &= \frac{x^Tx}{\left\lVert x \right\rVert _2} - 2v^T \frac{x}{\left\lVert x \right\rVert _2} + v^Tv \\
        &=2\left( 1 - v^T \frac{x}{\left\lVert x \right\rVert _2} \right) \\
        &= 2(1-c_{i})
    \end{align*}

    \[
        \left| \sum_{j = 1, j\neq 0}^{n} c_{j}^2  \right| = |1-c_{i}^2| = |(1-c_{i})(1+c_{i})| \leqslant 2|1-c_{i}| \leqslant 2C^2\epsilon^2
    \]

    \begin{align*}
        \left| R\left(\frac{x}{\left\lVert x \right\rVert _2} - \lambda_{i}\right) \right|
        &= \left|\frac{(x/\left\lVert x \right\rVert _2)^TA(x/\left\lVert x \right\rVert _2)}{(x/\left\lVert x \right\rVert _2)^T(x/\left\lVert x \right\rVert _2)} - \lambda_{i}\right| \\
        &= \left|(\sum_{j=1}^{n} c_{i} )^TA(\sum_{j=1}^{n} c_{i} ) - \lambda_{i} \right| \\
        &= \left|\left(\sum_{j=1}^{n} \lambda_{j}c^2_j \right) - \lambda_{i}\right| \\
        &= |\lambda_{i} (c_{i}^2 - 1) + \sum_{j=1, j \neq i}^{n} \lambda_{j}c_{j}^2 | \\
        &\leqslant |\lambda_{i}||1-c_{i}^2| + \max_{j \neq i}|\lambda_{j}|(\sum_{i=1}^{n}  )
    \end{align*}

\end{proof}


\begin{theorem}
    
    $ A \in \mathbb{R}^{n \times n}_{\text{sym}}, v \in \mathbb{R}^n $, with
    \[
        (A - \mu I) v = w
    \]
    then There is an eigenvalue $ \lambda_{i} $ such that $ \mu - \lambda_{i} \leqslant \frac{\left\lVert w \right\rVert _2}{\left\lVert v \right\rVert _2} $ 

\end{theorem}

\begin{proof}
    
    $ A \in \mathbb{R}^{n \times n}_{\text{sym}} $, $ (q_{i})_{i=1}^{n} $ orthonormal basis of eigenvectors.
    \[
        v = \sum_{i=1}^{n} c_{i}q_{i}, \hspace{1em} w = \sum_{i=1}^{n} d_{i}q_{i}
    \]

    \begin{align*}
        (A - \mu I)(\sum_{i=1}^{n} c_{i}q_{i} )
        &= \sum_{i=1}^{n} c_{i}Aq_{i} - \sum_{i=1}^{n}  c_{i}\mu q_{i} \\
        &= \sum_{i=1}^{n} c_{i}\lambda_{i} q_{i} - \sum_{i=1}^{n}  c_{i}\mu q_{i}\\
        &= \sum_{i=1}^{n} c_{i} (\lambda_{i}-\mu) q_{i} \\
    \end{align*}

    So $ (\lambda_{i} - \mu )c_{i} = d_{i} $. $ c_{i} = \frac{1}{\lambda_{i} - \mu} d_{i}$ assuming $ \lambda_{i} \neq \mu $ for all $ i $.

    \[
        \left\lVert v \right\rVert _2^2 = \left|\sum_{i=1}^{n} c_{i}^2 \right| 
        = \left|\sum_{i=1}^{n} \frac{1}{\lambda_{i} - \mu} d_{i}^2 \right| 
        \leqslant \max_{i} \frac{1}{(\lambda_{i} - \mu)^2} \cdot \sum_{i=1}^{n} d_{i}^2 
        = \max_{i} \frac{1}{(\lambda_{i} - \mu)^2} \cdot \left\lVert w \right\rVert _2^2
    \]

\end{proof}
