\begin{theorem} \underbar{\textbf{The Secant Method}}

    \[
        x_{k+1} = x_{k} - \frac{x_{k}-x_{k-1}}{f(x_{k}) - f(x_{k-1})} f(x_{k})
    \]

    Suppose $ f: [a,b] \to \mathbb{R}$ continuous satisfies
    \begin{itemize}
        \item $ f' $ cts in a neighbourhood of $ \xi $ \hspace{2em}($ f(\xi) = 0 $)
        \item $ f'(\xi) \neq 0 $
    \end{itemize}

    The secant method converges to $ \xi $, if $ x_0 $ and $ x_1 $ are sufficiently closed to $ \xi $.
    
\end{theorem}

    \]

\begin{definition} \leavevmode
    
    \noindent\underbar{\textbf{Vector Space}} or \underbar{\textbf{Linear Space}} has two binary operations
    \begin{enumerate}
        \item scalar multiplication: $ \mathbb{R} \times V \to V $
        \item vector addition: $ V \times V \to V $
    \end{enumerate}

\end{definition}

\begin{theorem} properties of vector space \leavevmode
    
    \begin{itemize}
        \item $ a(b \overrightarrow{u}) = (ab) \overrightarrow{u}$
        \item $ \overrightarrow{u} + \overrightarrow{v} = \overrightarrow{v} + \overrightarrow{u}$
        \item $ (\overrightarrow{u} + \overrightarrow{v}) + \overrightarrow{w} = \overrightarrow{u} + (\overrightarrow{v} + \overrightarrow{w})$
        \item $ a(\overrightarrow{u} + \overrightarrow{v}) = a\overrightarrow{u} + a\overrightarrow{v}$
        \item $ (a+b)\overrightarrow{u} = a\overrightarrow{u} + b \overrightarrow{u} $
        \item $ \overrightarrow{0} \in V $ such that $ \overrightarrow{u} + \overrightarrow{0} = \overrightarrow{u} $
        \item $ 1\overrightarrow{u} = \overrightarrow{u} $
        \item for each $ \overrightarrow{u} \in V $, there is $ -\overrightarrow{u} \in V$ such that $ (\overrightarrow{u}) + (-\overrightarrow{u}) = 0 $
    \end{itemize}
\end{theorem}

\vspace{2em}
\begin{definition} Norm \leavevmode
    
    Let $ V $ be a vector space, $ \left\lVert  \right\rVert : V \to \mathbb{R}_{+} $.

    \begin{itemize}
        \item $ \left\lVert \overrightarrow{v} \right\rVert \iff \overrightarrow{v} = 0  \overrightarrow{v} \in V $
        \item $ \left\lVert \lambda \overrightarrow{u} \right\rVert = |\lambda | \left\lVert \overrightarrow{u} \right\rVert $, $ \lambda  \in \mathbb{R}, \overrightarrow{u} \in V $
        \item $ \left\lVert \overrightarrow{u} + \overrightarrow{v} \right\rVert \leqslant  \left\lVert \overrightarrow{u} \right\rVert + \left\lVert \overrightarrow{v} \right\rVert$
    \end{itemize}

\end{definition}

\begin{definition} 1-norm, 2-norm, $ \infty $ norm \leavevmode 
    
    Let $ \overrightarrow{x} \in \mathbb{R}^n $

    \begin{itemize}
        \item $ \left\lVert x \right\rVert _{1} = \sum_{i=1}^{n}|x_{i}| $
    \end{itemize}


    
\end{definition}
