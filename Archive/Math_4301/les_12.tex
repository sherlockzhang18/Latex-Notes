\begin{definition}
    Dual Basis: 

    $ v_1, \cdots , v_{n} $ a basis for $ V $, then we pick a dual basis $ \phi_1, \cdots , \phi_{n} $ such that 

    \[
        \phi_{i}(v_{j}) = \begin{cases}
            0 \hspace{2em} j \neq i \\
            1 \hspace{2em} j = i
        \end{cases}
    \]
\end{definition}

\begin{eg}
    
    \begin{enumerate}
        \item     
    $ T: \mathbb{R}^2 \to \mathbb{R}^2 $ 
    $ T(x, y) =  (x - y, x + y) $

    $ T'(\phi) = \phi \circ T = \phi(x - y, x + y) $

    \vspace{1em}
    A linear function on $ \mathbb{R}^2 $ $ \phi \in \mathbb{R}^2 $

    \[
        \phi(x, y) = ax + by
    \]
    \[
        \phi(x(1,0), y (0,1)) = x\phi(1,0) = y \phi(0,1)
    \]

    If $ \phi(x, y) = ax + by $, then $ T'(\phi) (x, y) = \phi(x - y, x + y) $.

    \item 
    \end{enumerate}

\end{eg}

\vspace{2em}
\begin{definition} Annihilators \leavevmode

    If $ U $ is a subspace of $ V $, we get a subspace of $ V' $.

    Annihilator of $ U $: $ U^0 = \{\phi \in V' | \phi(v) = 0 \text{ for all } v \in U\} $

\end{definition}

\vspace{1em}
\begin{eg}
    
    \begin{itemize}
        \item $ \{0\}^0 = V' $
        \item $ V^0 = 0 \to  $ the zero functional
        
        \item $ V = \mathbb{R}^2, U = \{(\lambda, \lambda) | \lambda \in \mathbb{R}^2\} $, then $ U^0 = \{\phi | \phi(\lambda, \lambda) = 0 \text{ for all } \lambda\} =  \{\phi | a\lambda + b\lambda = 0 \text{ for all } \lambda\} = \{\phi | \phi(x, y) = ax - ay\} $.
    \end{itemize}

\end{eg}

\begin{note}
    
    \begin{enumerate}
        \item $ U^0 $ is a subspace of $ V' $.
        \item If $ U $ is a subspace of $ V $, then 
        \[
            \dim U + \dim U^0 = \dim V
        \]
    \end{enumerate}

\end{note}
