\underbar{\textbf{Eigenvalue Problems:}}

\begin{definition} \leavevmode
    
    $ A \in \mathbb{R}^{n \times n}_\text{sym}, \lambda \in \mathbb{R}, x \in \mathbb{C}^n \backslash \{0\} $ such that 
    \[
        Ax = \lambda x \iff (A - \lambda I)x = Bx = 0
    \]
    where $ \lambda_{i} $ are eigenvalues, and $ x $ are eigenvectors.

\end{definition}

\vspace{1em}
\begin{itemize}
    \item 
        Characteristic Polynomial:
        \[
            P(\lambda) = \text{det}(A - \lambda I): \text{deg } n \text{ polynomial}
        \]

        \vspace{1em}
        LL is eigenvalue $ \iff $ $ P(\lambda) = 0 $

    \item 
        Characteristic Polynomial is invariant under similarity transformations
        \[
            A = XBX^{-1}
        \]
        where $ X $ is inverse coordinate transformation and $ X^{-1} $ is coordinate transformation.

        \vspace{1em}
        \begin{align*}
            P(\lambda) 
            &= \text{det } (XBX^{-1} - \lambda XX{-1}) \\
            &= \text{det } (X(B - \lambda I)X^{-1}) \\
            &= \text{det}(X ) \text{det}(B - \lambda I) \text{det}(X^{-}1)
        \end{align*}

    \item use similarity transforms so $ B $ is easy
    \item Problem of find eigenvalues is equivalent to problem of finding roots to polynomials
    \item There is no closed form solution for finding roots of polynomials of $ \text{deg } n \geqslant 5 $. (Abel-Ruffini's Theorem)
    \item Need iterative algorithm for $ n \geqslant 5 $.
\end{itemize}

\vspace{1em}
\begin{theorem}
    
    $ A \in \mathbb{R}^{n \times n}_{\text{sym}}, A = Q \Lambda Q^T $ ($ \Lambda $ diagonal and $ Q $ orthogonal, $ Q^TQ = QQ^T = I $)
    then
    \begin{align*}
        &Aq_{i} = \lambda_{i} q_{i} \\
        A\left[q_1 | q_2 | \cdots | q_{n}\right] = 
        &\left[q_1 | q_2 | \cdots | q_{n}\right] 
        \begin{bmatrix}
          \lambda_1 & 0 & 0 & 0 \\
          0 & \lambda_2 & 0 & 0 \\
          0 & 0 & \ddots & \vdots \\
          0 & 0 & \cdots & \lambda_{n} \\
        \end{bmatrix} \\
    \end{align*}
    , where $ \lambda_{i} $ eigenvalues, $ q_{i} $ eigenvectors, $ \{q_{i}\}_{i = 1}^{n} $ orthonormal

    \[
        AQ = QA, A = QAQ^T
    \]

\end{theorem}

\vspace{1em}
\underbar{\textbf{Goal:}} Find nice orthogonal transformations

\vspace{1em}
\begin{definition} (Rotation Matrix) \leavevmode
    
    $ R^{(\varphi)}_{pq} \in \mathbb{R}^{n \times n} $

    \[
        [R^{(\varphi)}_{pq}]_{ij} = 
        \begin{cases}
            \cos(\varphi) \text{ if } i = p \text{ and } j = p \\
            \sin(\varphi) \text{ if } i = p \text{ and } j = q \\
            \cos(\varphi) \text{ if } i = q \text{ and } j = q \\
            -\sin(\varphi) \text{ if } i = q \text{ and } j = p \\
            \delta_{ij} \text{ if } i \neq p \text{ or } q \text{ or } j \neq p \text{ or } q
        \end{cases}
    \]

\end{definition}

\vspace{1em}
\begin{theorem} \leavevmode
    
    $ A \in \mathbb{R}^{n \times n}_{\text{sym}} $, $ B = R^{(\varphi)T}_{pq} A R^{(\varphi)T}_{pq} $

    \vspace{1em}
    then there exists $ \varphi $ such that $ b_{pq} = [B]_{pq} = 0 $.

\end{theorem}

\vspace{1em}
\begin{proof} \leavevmode
    
    $ A' = A R^{(\varphi)T}_{pq} $
    \[
        \left[a_1'|a_2'| \cdots | a_{n}'\right] = \left[a_1|a_2| \cdots | a_{n}\right] R^{(\varphi)T}_{pq}
    \]

    then,
    \begin{align*}
        a_{p}' = ca_{p} - sa_{q} \\
        a_{q}' = sa_{p} + ca_{q} \\
    \end{align*}

    $ B = R^{(\varphi)T}_{pq} A' $:
    \begin{align*}
        b_{pp} 
        &= c(ca_{pp} - sa_{pq}) - s(ca_{qp} - sa_{qq}) \\
        &= c^2a_{pp} - sca_{pq} + sca_{qp} + s^2a_{qq} \\ 
        &c^2a_{pp} - 2sca_{pq} + s^2a_{qq} \\ \\
        b_{pq} = b_{qp}
        &= sa_{p} + ca_{q} \\
        &= s(ca_{pp} - sa_{pq}) + c(ca_{qp} - sa_{qq})\\
        &= sca_{pp} - s^2a_{pq} + c^2a_{qp} - sca_{qq} \\
        &= sca_{pp} + (c^2 - s^2) a_{pq} - sca_{qq}\\
        &= sc(a_{pp} - a_{qq}) + (c^2 - s^2)a_{pq} \\
        &= \frac{1}{2}\sin(2 \varphi)(a_{pp} - a_{qq}) + \cos(2\varphi) a_{pq}\\
    \end{align*}

    \vspace{1em}
    Next is to find $ \varphi $ such that $ b_{pq} = 0 $

    \[
        b_{pq} = v \cdot w \text{ where } 
        \begin{cases}
            v = (-\sin(2\varphi), \cos(2\varphi)) \\
            w = (\frac{a_{qq} - a_{pp}}{2}, a_{pq})
        \end{cases}
    \]

    \[
        2\varphi = \frac{1}{2} \arctan 2 \left(a_{pq}, \frac{(a_{qq} - a_{pp})}{2} \right),
    \]
    then $ b_{pq} = 0 $

\end{proof}
