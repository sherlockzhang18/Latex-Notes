\begin{definition} Frobenius Norm
    
    Let $ A \in \mathbb{R}^{m \times n} $
    \[
        \left\lVert A \right\rVert _F = \left(\sum_{j = 1}^{n} \sum_{i = 1}^{m} |a_{ij}|^2\right)^{\frac{1}{2}}
    \]

    \vspace{1em}
    Remark: F.norm is different from subordinate matrix norms (operator norms).

\end{definition}

\vspace{2em}
\begin{lemma} \leavevmode
    
    $ A \in \mathbb{R}^{m \times n}, Q_{node} \in \mathbb{R}^{n \times n}, Q_{m} \in \mathbb{R}^{m \times n} $, where $ Q_{n} $ and $ Q_{m} $ orthogonal

    \begin{itemize}
        \item $ \left\lVert A \right\rVert _F = \left\lVert A^T \right\rVert _F $
        \item $ \left\lVert Q_{m}A \right\rVert _F = \left\lVert A \right\rVert _F $
        \item $ \left\lVert A \right\rVert _F = \text{tr}(A^TA)^{\frac{1}{2}} $
    \end{itemize}

\end{lemma}

\vspace{1em}
\begin{proof} \leavevmode
    
    \[
        (A^TA)_{ii} = \sum_{j = 1}^{m} a_{ji}a_{ji} = \sum_{j=1}^{m} |a_{ji}|^2 
    \]

    \[
        \text{tr}(A^TA) = \sum_{j=1}^{m} (A^TA)_{ii} = \sum_{i=1}^{n} \sum_{j=1}^{m} |a_{ji}|^2
    \]

    \[
        \left\lVert Q_{m}A \right\rVert _F = \text{tr}(A^TQ_{m}^TQ_{m}A)^{\frac{1}{2}} = \left\lVert A \right\rVert _F
    \]

    remark: $ R^{(\varphi)T}_{pq} R^{(\varphi)T}_{pq} = I $ orthogornal.

\end{proof}

\vspace{2em}
\begin{definition} Jacobi's Method \leavevmode
    
    $ A \in \mathbb{R}^{n \times n}_{\text{sym}} $, $ A^{(0)}: = A $

    \vspace{1em}
    Step 1: (Find the largest off diagonal entry of $ A^{(k)} $). Find $ p, q $ $ 1 \leqslant p < q \leqslant n $ such that
    \[
        |a_{pq}^{(k)}| \geqslant |a_{ij}^{(k)}| \text{ for } i \neq j
    \]

    \vspace{1em}
    Step 2: (Eliminate $ a_{pq}^{(k)} $ using $ R_{pq}^{(\varphi)} $ ). Find $ \varphi $ such that
    \[
        A^{(k + 1)} = R^{(\varphi)T}_{pq} A^{(k)} R^{(\varphi)T}_{pq}, \hspace{2em} a_{pq}^{(k+1)} = 0
    \]

    \vspace{1em}
    Remark: Zeros introduced in $ (k-1) $st iteration is not necessarily preserved (doesn't stay zero) in the k-th iteration.

\end{definition}

\vspace{1em}
\begin{proof}
    
    Let $ D(A) = \sum_{i=1}^{n} |a_{ii}|^2 $, $ L(A) = 2 \sum_{j=i+1}^{n} \sum_{i=1}^{n} |a_{ij}|^2 $
    then,
    \[
        \left\lVert A \right\rVert _F^2 = D(A) + L(A)
    \]

    \vspace{1em}
    Let $ B $ be the next iterate, $ \left\lVert B \right\rVert ^2_F = D(B) + L(B) $

    \[
        \begin{cases}
            D(B) = D(A) + 2a_{pq}^2 \\
            L(B) = L(A) - 2a_{pq}^2
        \end{cases}
    \]
    since $ a_{pp}^2 + a^2_{qq} + 2a^2_{pq} = b^2_{pp} + b_{qq}^2 $.

    \vspace{1em}
    $ |a_{pq}|^2 \leqslant \frac{1}{\text{number offdiagonal}} L(A) = \frac{1}{n(n-1)} L(A) $

    \begin{align*}
        L(B) 
        &= L(A) - 2|a_{pq}|^2 \\
        &\leqslant \left(1 - \frac{1}{n(n-1)}\right) L(A) \\
    \end{align*}
    where $ \alpha = \left(1 - \frac{1}{n(n-1)}\right) $ and $ 0 \leqslant \alpha <1 $
    
    so $ L(A)^{(k+1)} \leqslant \alpha^k L(A) $ hence 

\end{proof}
